### 条件判断

```kotlin
fun 单条件判断(甲: 整型, 乙: 整型) {

    var 癸: 字符串?
    // 条件判断 则 表达式｜{ 代码块 } <否则 表达式｜{ 代码块 }>
    癸 = 甲 等于 乙 则 "甲等于乙" 否则 "甲不等于乙"
    癸.打印换行()
    // 如果(条件判断) <则 表达式>｜<则> { 代码块 } <否则 表达式｜{ 代码块 }>
    癸 = 如果(甲 等于 乙) { "甲等于乙" } 否则 "甲不等于乙"
    癸.打印换行()
    癸 = 如果(甲 等于 乙) 则 "甲等于乙" 否则 "甲不等于乙"
    癸.打印换行()
    // 如果 { 代码块 } 则 表达式｜{ 代码块 } <否则 表达式｜{ 代码块 }>
    癸 = 如果 { 甲 等于 乙 } 则 "甲等于乙" 否则 "甲不等于乙"
    癸.打印换行()

    换行打印()

}
```

```kotlin
fun 多条件判断(甲: 整型, 乙: 整型) {

    var 癸: 字符串?

    // 如果(条件判断) { 代码块 }.或者(条件判断) { 代码块 }.否则 {}
    // 如果(条件判断) { 代码块 }.或者(条件判断) { 代码块 } 否则 {}
    癸 = 如果(甲 等于 1) {
        "甲等于1"
    }.或者(乙 等于 2) {
        "甲等于2"
    }.或者(甲 等于 乙) {
        "甲等于乙"
    }.否则 {
        "甲不等于乙"
    }
    癸.打印换行()

    // 如果(条件判断) { 代码块 } ?: 如果(条件判断) { 代码块 } ?: 最后 {}
    // 如果(条件判断) { 代码块 } ?: 如果(条件判断) { 代码块 } 否则 {}
    癸 = 如果(甲 等于 1) {
        "甲等于1"
    } ?: 如果(乙 等于 2) {
        "甲等于2"
    } ?: 如果(甲 等于 乙) {
        "甲等于乙"
    } ?: 最后 {
        "甲不等于乙"
    }
    癸.打印换行()

    换行打印()

}
```

### 循环

```kotlin
fun 区间循环() {

    结束 {

        // 0 到之前 10 每隔 2 循环
        // 0 到 10.之前 每隔 2 循环
        // 10 减到 0  每隔 1 循环 {}
        0 到 10 每隔 2 循环 { 它 ->
            // 跳过当前循环
            如果(它 等于 4) 则 { return@循环 }
            // 结束循环
            如果(它 大于等于 7) { return@结束 }
            它.打印换行()
        }
    }
}
```

```kotlin
fun 条件循环() {

    var 它 = 0
    // 当 { 条件判断 } 循环 { 代码块 }
    // 循环 { 代码块 } 当 { 条件判断 }
    当 { 它 < 5 } 循环 {
        它.打印换行("前置判断 ")
        它++
    }

    换行打印()

    循环 {
        它.打印换行("后置判断 ")
        它++
    } 当 { 它 小于 10 }

}
```
